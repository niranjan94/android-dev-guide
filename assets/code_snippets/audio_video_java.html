/// PLAY AUDIO FROM PACKAGE
///
MediaPlayer AudioInternal = MediaPlayer.create(getBaseContext(), R.raw.sample); // KEEP AUDIO FILE IN raw FOLDER OF PACKAGE
AudioInternal.start(); // START PLAYING
AudioInternal.stop();  // STOP PLAYING
///
/// PLAY AUDIO FROM EXTERNAL SOURCE (Internet)
///
MediaPlayer AudioExternal = new MediaPlayer();
AudioExternal.setDataSource("http://njlabs.kovaideals.com/api/agad/sample.mp3");
AudioExternal.prepare();
AudioInternal.start(); // START PLAYING
AudioInternal.stop();  // STOP PLAYING
///
/// PLAY VIDEO FILE
///
getWindow().setFormat(PixelFormat.TRANSLUCENT);
VideoView videoPlayer = (VideoView) findViewById(R.id.videoView);
videoPlayer.setMediaController(new MediaController(this)); // TO ADD STANDARD MEDIA CONTROLS
Uri video = Uri.parse("android.resource://"+getPackageName()+"/"+R.raw.sample_video); // FOR VIDEO FROM PACKAGE (raw) 
// Uri video = Uri.parse(path+"/"+fileName); // IF VIDEO IS SOMEWHERE ELSE (like in memory card)
videoPlayer.setVideoURI(video);
videoPlayer.start(); // START PLAYING
videoPlayer.stopPlayback(); // STOP PLAYING

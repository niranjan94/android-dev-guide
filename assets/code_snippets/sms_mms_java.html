///
/// OPEN SMS APPLICATION WITH THE NUMBERAND MESSAGE ALREADY ENTERED
///
String PhoneNumber="+919600514966";
String Message="The Message to be sent";
Intent SmsIntent = new Intent(Intent.ACTION_VIEW);
SmsIntent.putExtra("address", Message);
SmsIntent.putExtra("sms_body", PhoneNumber); 			
SmsIntent.setType("vnd.android-dir/mms-sms");
startActivity(SmsIntent); 
///
/// OPEN AN MMS APPLICATION WITH THE NUMBER, MEDIA FILE AND MESSAGE ALREADY ENTERED
///
String PhoneNumber="+919600514966";
String Message="The Message to be sent";
String MediaFile="/sdcard/image.png";  // PATH TO THE MEDIA FILE TO BE ATTACHED
Intent MmsIntent = new Intent(Intent.ACTION_SEND);
MmsIntent.putExtra("address", PhoneNumber);
MmsIntent.putExtra("sms_body", Message);			 
File img = new File(MediaFile);  // OPEN MEDIA FILE TO ATTACH IT
MmsIntent.putExtra(Intent.EXTRA_STREAM, Uri.fromFile(img));  
MmsIntent.setType("image/png");
startActivity(MmsIntent);

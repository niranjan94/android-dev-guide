/// STORING TO SHARED PREFERENCES
///
SharedPreferences preferences = getSharedPreferences("pref_file_name", Context.MODE_PRIVATE);
SharedPreferences.Editor editor = preferences.edit();
editor.putString("Key1", "Value to store"); // FOR A STRING VALUE
editor.putBoolean("Key2", true); // FOR A BOOLEAN (true/false) VALUE
editor.putFloat("Key3", 1545); // FOR A FLOAT VALUE
editor.putInt("Key4", -1545); // FOR AN INTEGER VALUE
editor.commit();
///
/// READING FROM SHARED PREFERENCES
///
SharedPreferences preferences = getSharedPreferences("pref_file_name", Context.MODE_PRIVATE);
Variable1 = preferences.getString("Key1", null); // getString(key_name, default_value) // FOR STRING
Variable2 = preferences.getBoolean("Key2", true); // getBoolean(key_name, default_value) // FOR BOOLEAN
Variable3 = preferences.getInt("Key3", 0); // getString(key_name, default_value) // FOR FLOAT
Variable4 = preferences.getFloat("Key4", 0); // getString(key_name, default_value) // FOR INT

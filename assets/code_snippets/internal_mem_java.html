/// STORING TO INTERNAL MEMORY
///
FileOutputStream fs = openFileOutput("data_file.txt", Context.MODE_PRIVATE);  // OPEN A FILE STREAM
fs.write("Data to Store".getBytes()); // WRITE DATA TO THE FILE TREAM
fs.flush(); // FLUSH THE FILE STREAM
fs.close(); // CLOSE FILE STREAM
///
/// READING FROM INTERNAL MEMORY
///
String RetrievedString;
int ch;
StringBuilder str = new StringBuilder();
FileInputStream fs = openFileInput("data_file.txt");  // OPEN A FILE STREAM
while ((ch = fs.read()) != -1)  // READ CHARACTER BY CHARACTER FROM THE FILE TREAM
{
	str.append((char) ch); 
}
fs.close(); // CLOSE FILE STREAM
RetrievedString = str.toString();

/// A CUSTOM SPEAKER FUNCTION
///
String TextToSpeak;
public void Speak(String Text)
{
    TextToSpeak=Text;
    talker = new TextToSpeech(getBaseContext(), new OnInitListenerTTS());
}
class OnInitListenerTTS implements OnInitListener
{
    @Override
    public void onInit(int status) {
        talker.speak(TextToSpeak, TextToSpeech.QUEUE_FLUSH, null);
    }
    public void onDestroy() {
        if (talker != null) {
            talker.stop();
            talker.shutdown();
        }
    }
}
///
/// TO USE THE SPEAKER FUNCTION
///
/// Just add the Function, Class given above and Call the following function with the Text
///
Speak("Text to Speak out");
